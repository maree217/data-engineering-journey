Python Data Analysis

                            

                                
import pandas as pd
import numpy as np

# Load data
df = pd.read_csv('customer_data.csv')

# Basic statistics
print(df.describe())

# Group by city
city_analysis = df.groupby('city')['purchase_amount'].agg([
    'mean', 'sum', 'count'
])

print(city_analysis)
